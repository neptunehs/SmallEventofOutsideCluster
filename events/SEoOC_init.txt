namespace = nep_SEoOC_init

#Diff
country_event = {
	id = nep_SEoOC_init.1
	hide_window = yes
	
	fire_only_once = yes

	is_triggered_only = yes
	
	trigger = {
		is_ai = no
		NOT = { has_global_flag = nep_SEoOC_active }
	}
	
	immediate = {
		set_global_flag = nep_SEoOC_active
		if = {
			limit = { NOT = { has_global_flag = has_nsc_active } }
			set_global_flag = nep_SEoOC_easy
			every_country = {
				limit = { can_change_policy = nep_SEoOC_diff_policy }
				set_policy = {
					policy = nep_SEoOC_diff_policy
					option = SEoOC_diff_diff_policy_easy
					cooldown = no
				}
			}
		}
		else = {
			every_country = {
				limit = { can_change_policy = nep_SEoOC_diff_policy }
				set_policy = {
					policy = nep_SEoOC_diff_policy
					option = SEoOC_diff_diff_policy_default
					cooldown = no
				}
			}
		}
	}
}
